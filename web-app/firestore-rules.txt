// Reglas de Firestore para MixerCurse Web App
// Copia y pega estas reglas en Firebase Console > Firestore > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Usuarios pueden leer/escribir solo sus propios datos
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Proyectos pueden ser leídos/escritos solo por su propietario
    match /projects/{projectId} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.ownerId;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.ownerId;
    }
    
    // Setlists pueden ser leídos/escritos solo por su propietario
    match /setlists/{setlistId} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.ownerId;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.ownerId;
      
      // Canciones dentro de setlists
      match /songs/{songId} {
        allow read, write: if request.auth != null && 
          request.auth.uid == get(/databases/$(database)/documents/setlists/$(setlistId)).data.ownerId;
        allow create: if request.auth != null && 
          request.auth.uid == get(/databases/$(database)/documents/setlists/$(setlistId)).data.ownerId;
      }
    }
  }
}
